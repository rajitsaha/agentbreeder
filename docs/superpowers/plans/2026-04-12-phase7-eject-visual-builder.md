**Goal:** Generate framework-native eject scaffolds for CrewAI/ADK/Claude SDK, persist builder sessions to disk, and show framework-appropriate node palettes in the visual builder.

**Architecture:** The eject command dispatches on framework type to generate authentic scaffold files matching each framework's official project structure. Builders API uses a file-backed JSON store keyed by `{resource_type}/{name}`. The visual builder's `NodePalette` component accepts a `framework` prop and renders the appropriate node set from a per-framework palette map.

**Tech Stack:** Python Typer CLI, FastAPI, React 18 + TypeScript, ReactFlow

```python
def _generate_crewai_scaffold(yaml_content: str, out_dir: Path) -> None:
    """Generate a CrewAI-native project scaffold from agent.yaml."""
    import yaml

    data = yaml.safe_load(yaml_content)
    if not isinstance(data, dict):
        raise typer.BadParameter("Invalid YAML: expected a mapping at the top level")

    name = data.get("name", "my-agent")
    description = data.get("description", f"A CrewAI agent for {name}")
    model_cfg = data.get("model", {})
    primary_model = model_cfg.get("primary", "gpt-4o") if isinstance(model_cfg, dict) else "gpt-4o"
    tools = data.get("tools", [])
    tool_names = []
    for t in tools:
        if isinstance(t, dict):
            tool_names.append(t.get("name") or t.get("ref", "unknown_tool").split("/")[-1])

    # --- crew.py ---
    tool_import_lines = "\n".join(
        f"    # from crewai_tools import {tn.replace('-', '_').title()}Tool  # TODO: import real tool"
        for tn in tool_names
    ) or "    # No tools defined — add imports here"

    tool_assignments = "\n".join(
        f"        {tn.replace('-', '_').replace('/', '_')} = {tn.replace('-', '_').replace('/', '_').title()}Tool()"
        for tn in tool_names
    ) or "        # tools = [MyTool()]"

    tool_list = (
        "[" + ", ".join(tn.replace("-", "_").replace("/", "_") for tn in tool_names) + "]"
        if tool_names
        else "[]"
    )

    crew_py = f'''\
"""CrewAI agent scaffold generated by `agentbreeder eject`.

Edit this file to customise roles, goals, backstories, and task logic.
Deploy with: agentbreeder deploy agent.yaml
"""

from crewai import Agent, Task, Crew, Process
from crewai.project import CrewBase, agent, crew, task

{tool_import_lines}


@CrewBase
class {_to_class_name(name)}Crew:
    """{description}"""

    agents_config = "config/agents.yaml"
    tasks_config = "config/tasks.yaml"

    # ------------------------------------------------------------------
    # Agents
    # ------------------------------------------------------------------

    @agent
    def primary_agent(self) -> Agent:
        return Agent(
            config=self.agents_config["primary_agent"],  # type: ignore[index]
            llm="{primary_model}",
            verbose=True,
        )

    # ------------------------------------------------------------------
    # Tasks
    # ------------------------------------------------------------------

    @task
    def primary_task(self) -> Task:
        return Task(
            config=self.tasks_config["primary_task"],  # type: ignore[index]
        )

    # ------------------------------------------------------------------
    # Crew
    # ------------------------------------------------------------------

    @crew
    def crew(self) -> Crew:
        return Crew(
            agents=self.agents,   # populated by @agent decorators
            tasks=self.tasks,     # populated by @task decorators
            process=Process.sequential,
            verbose=True,
        )
'''

    # --- config/agents.yaml ---
    agents_yaml = f'''\
primary_agent:
  role: >
    Primary agent for {name}
  goal: >
    Complete the assigned tasks accurately and efficiently
  backstory: >
    You are a highly capable AI assistant specialised in {description.lower()}.
    You have deep expertise in your domain and always deliver high-quality results.
'''

    # --- config/tasks.yaml ---
    tasks_yaml = f'''\
primary_task:
  description: >
    Execute the primary workflow for {name}.
    Use all available tools and information to complete the objective.
  expected_output: >
    A comprehensive, well-structured response that fully addresses the task.
  agent: primary_agent
'''

    # --- requirements.txt ---
    requirements_txt = "crewai>=0.80\ncrewai-tools>=0.4\n"

    # Write all files
    out_dir.mkdir(parents=True, exist_ok=True)
    (out_dir / "crew.py").write_text(crew_py, encoding="utf-8")
    config_dir = out_dir / "config"
    config_dir.mkdir(exist_ok=True)
    (config_dir / "agents.yaml").write_text(agents_yaml, encoding="utf-8")
    (config_dir / "tasks.yaml").write_text(tasks_yaml, encoding="utf-8")
    (out_dir / "requirements.txt").write_text(requirements_txt, encoding="utf-8")
```

```python
def _generate_google_adk_scaffold(yaml_content: str, out_dir: Path) -> None:
    """Generate a Google ADK-native project scaffold from agent.yaml."""
    import yaml

    data = yaml.safe_load(yaml_content)
    if not isinstance(data, dict):

    name = data.get("name", "my-agent")

    tools = data.get("tools", [])

        system_prompt = prompts.get("system", "")
```

```python
# ---------------------------------------------------------------------------
# File-backed store (survives restarts)
# ---------------------------------------------------------------------------

import os


class FileStore:
    """Persist builder YAML sessions as individual files under a base directory.

    Directory layout:
        {base_dir}/{resource_type}/{name}.yaml

    The base directory defaults to ~/.agentbreeder/builders/ and can be
    overridden with the BUILDERS_DATA_DIR environment variable.
    """

    def __init__(self) -> None:
        self._base = Path(
            os.environ.get("BUILDERS_DATA_DIR", Path.home() / ".agentbreeder" / "builders")
        )

    def _path(self, resource_type: str, name: str) -> Path:
        return self._base / resource_type / f"{name}.yaml"

    def get(self, resource_type: str, name: str) -> str | None:
        p = self._path(resource_type, name)
        if not p.exists():
            return None
        return p.read_text(encoding="utf-8")

    def set(self, resource_type: str, name: str, yaml_content: str) -> None:
        p = self._path(resource_type, name)
        p.parent.mkdir(parents=True, exist_ok=True)
        p.write_text(yaml_content, encoding="utf-8")

    def exists(self, resource_type: str, name: str) -> bool:
        return self._path(resource_type, name).exists()


_store = FileStore()
```

```tsx
                className={cn(
                  "flex cursor-grab items-center gap-2.5 rounded-lg border px-3 py-2 transition-colors active:cursor-grabbing",
```
